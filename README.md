* $apt update && apt upgrade

* $pkg install nodejs
* $pkg install git
* $pkg install bash
* $pkg install mc(buat setting ownerNumber)
* $git clone https://github.com/Finox999/self-bot1
* $cd self-bot1
* $npm install/bash install.sh(bebas yg mana aja)
* $npm start/node index.js(bebas yg mana aja)
